<head>
      <style>
            #wrap{
                border: 1px solid black;
            }
            #ulul li{
                display: inline-block; /* 가로 배치 */
                border: 1px solid black;
                margin-left: 10px;
                padding: 0 5px;
                list-style: none; /* li 앞에 점 없애기 */
            }
      </style>
</head>
<body>
    <div id="wrap">
        <ul id="ulul">
            <li>메일</li>
            <li>카페</li>
            <li>블로그</li>

    </div>
</body>
